/**
 * _apply-migration-113.ts — Applies migration 113 (judicial compass topics + stances + roles)
 * Run: cd C:\EV-Accounts && npx tsx backend/scripts/_apply-migration-113.ts
 */
import 'dotenv/config';
import fs from 'fs';
import path from 'path';
import pg from 'pg';

const { Pool } = pg;

const EXPECTED_TOPICS = 8;
const EXPECTED_STANCES = 40;
const EXPECTED_ROLES = 8;
const JUDICIAL_TOPIC_KEYS = [
  'judicial-criminal-justice',
  'judicial-access-to-justice',
  'judicial-government-deference',
  'judicial-transparency',
  'judicial-interpretation',
  'judicial-bail-pretrial',
  'judicial-prosecution-priorities',
  'judicial-police-accountability',
];

async function main(): Promise<void> {
  const databaseUrl = process.env['DATABASE_URL'];
  if (!databaseUrl) {
    console.error('ERROR: DATABASE_URL is not set.');
    process.exit(1);
  }

  const pool = new Pool({ connectionString: databaseUrl });
  try {
    // Pre-check
    const preCheck = await pool.query(
      `SELECT COUNT(*) FROM inform.compass_topics
       WHERE topic_key = ANY($1) AND is_live = true`,
      [JUDICIAL_TOPIC_KEYS]
    );
    if (parseInt(preCheck.rows[0].count) >= EXPECTED_TOPICS) {
      console.log(`SKIP — migration 113 already applied (${preCheck.rows[0].count} judicial topics exist)`);
      process.exit(0);
    }

    console.log('Applying migration 113...');
    const filePath = path.resolve(process.cwd(), 'backend', 'migrations', '113_judicial_compass_topics.sql');
    const sql = fs.readFileSync(filePath, 'utf-8');
    await pool.query(sql);
    console.log('Migration applied. Verifying...');

    // Verify topics
    const topicsCheck = await pool.query(
      `SELECT topic_key, judicial_role FROM inform.compass_topics
       WHERE topic_key = ANY($1) AND is_live = true
       ORDER BY created_at`,
      [JUDICIAL_TOPIC_KEYS]
    );
    console.log(`Topics: ${topicsCheck.rows.length}/${EXPECTED_TOPICS}`);
    if (topicsCheck.rows.length !== EXPECTED_TOPICS) {
      console.error(`FAIL — expected ${EXPECTED_TOPICS} topics, got ${topicsCheck.rows.length}`);
      process.exit(1);
    }
    for (const row of topicsCheck.rows) {
      console.log(`  ${row.topic_key} — judicial_role: ${row.judicial_role ?? 'NULL'}`);
    }

    // Verify stances
    const stancesCheck = await pool.query(
      `SELECT COUNT(*) FROM inform.compass_stances cs
       JOIN inform.compass_topics ct ON ct.id = cs.topic_id
       WHERE ct.topic_key = ANY($1)`,
      [JUDICIAL_TOPIC_KEYS]
    );
    const stanceCount = parseInt(stancesCheck.rows[0].count);
    if (stanceCount !== EXPECTED_STANCES) {
      console.error(`FAIL — expected ${EXPECTED_STANCES} stances, got ${stanceCount}`);
      process.exit(1);
    }
    console.log(`OK — stances: ${stanceCount}/${EXPECTED_STANCES}`);

    // Verify role rows — all must be 'judicial', none should be federal/state/local
    const rolesCheck = await pool.query(
      `SELECT role_scope, COUNT(*) FROM inform.compass_topic_roles ctr
       JOIN inform.compass_topics ct ON ct.id = ctr.topic_id
       WHERE ct.topic_key = ANY($1)
       GROUP BY role_scope`,
      [JUDICIAL_TOPIC_KEYS]
    );
    console.log('Role scope distribution:');
    for (const row of rolesCheck.rows) {
      console.log(`  ${row.role_scope}: ${row.count}`);
      if (row.role_scope !== 'judicial') {
        console.error(`FAIL — unexpected role_scope '${row.role_scope}' found for judicial topics`);
        process.exit(1);
      }
    }
    const totalRoles = rolesCheck.rows.reduce((sum, r) => sum + parseInt(r.count), 0);
    if (totalRoles !== EXPECTED_ROLES) {
      console.error(`FAIL — expected ${EXPECTED_ROLES} role rows, got ${totalRoles}`);
      process.exit(1);
    }
    console.log(`OK — role rows: ${totalRoles}/${EXPECTED_ROLES}, all 'judicial'`);

    // Verify isolation — no judicial topic has federal/state/local role rows
    const isolationCheck = await pool.query(
      `SELECT COUNT(*) FROM inform.compass_topic_roles ctr
       JOIN inform.compass_topics ct ON ct.id = ctr.topic_id
       WHERE ct.topic_key = ANY($1) AND ctr.role_scope IN ('federal', 'state', 'local')`,
      [JUDICIAL_TOPIC_KEYS]
    );
    if (parseInt(isolationCheck.rows[0].count) !== 0) {
      console.error('FAIL — judicial topics have non-judicial scope rows (scope isolation violated)');
      process.exit(1);
    }
    console.log('OK — scope isolation confirmed (0 federal/state/local rows for judicial topics)');

    console.log('All checks passed.');
    process.exit(0);
  } catch (err: unknown) {
    const message = err instanceof Error ? err.message : String(err);
    console.error('ERROR:', message);
    process.exit(1);
  } finally {
    await pool.end();
  }
}

main();
