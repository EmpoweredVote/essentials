```sql
-- 3. 2nd Middlesex Democratic Primary race + candidates
DO $$
DECLARE
  v_primary_id UUID;
  v_race UUID;
BEGIN
  SELECT id INTO v_primary_id FROM essentials.elections
  WHERE name = '2026 Massachusetts State Primary' AND state = 'MA';

  -- Insert race (primary_party required for primary races; unique on (election_id, position_name, primary_party))
  INSERT INTO essentials.races (election_id, office_id, position_name, primary_party, seats)
  VALUES (
    v_primary_id,
    'b1ed4e2a-4a9c-4b41-9e46-8500f608e026',  -- 2nd Middlesex district office_id (verified in DB)
    'MA State Senate 2nd Middlesex District',
    'Democratic',
    1
  )
  ON CONFLICT (election_id, position_name, primary_party) DO NOTHING
  RETURNING id INTO v_race;

  -- If already existed, fetch the id
  IF v_race IS NULL THEN
    SELECT id INTO v_race FROM essentials.races
    WHERE election_id = v_primary_id
      AND position_name = 'MA State Senate 2nd Middlesex District'
      AND primary_party = 'Democratic';
  END IF;

  -- 5 known Democratic primary candidates
  -- Azeem has an existing politician row (Cambridge City Councillor) — link via politician_id
  -- Others have no politician record yet — full_name only
  INSERT INTO essentials.race_candidates
    (race_id, full_name, first_name, last_name, politician_id, is_incumbent, candidate_status, source)
  VALUES
    (v_race, 'Burhan Azeem', 'Burhan', 'Azeem',
     'd2358e54-6860-4382-8c8d-95a3dabea874',  -- existing Cambridge Councillor politician_id
     false, 'active', 'ballotpedia'),           -- is_incumbent=false: open seat (Jehlen retiring)
    (v_race, 'Christine Barber', 'Christine', 'Barber',
     NULL, false, 'active', 'ballotpedia'),
    (v_race, 'Tom Hopcroft', 'Tom', 'Hopcroft',
     NULL, false, 'active', 'ballotpedia'),
    (v_race, 'Matt McLaughlin', 'Matt', 'McLaughlin',
     NULL, false, 'active', 'ballotpedia'),
    (v_race, 'Erika Uyterhoeven', 'Erika', 'Uyterhoeven',
     NULL, false, 'active', 'ballotpedia')
  ON CONFLICT DO NOTHING;
END $$;
```

CRITICAL pitfalls:
- Azeem is NOT the incumbent — this is an open seat; Patricia Jehlen is retiring. is_incumbent=false.
- The office_id b1ed4e2a-4a9c-4b41-9e46-8500f608e026 is the 2nd Middlesex (25D27) district office, verified in research.
- primary_party='Democratic' is required; without it the row falls under the partial index for general races.
- ON CONFLICT DO NOTHING on race_candidates uses the table's partial unique index on external_id WHERE external_id IS NOT NULL. Since we're not setting external_id, ON CONFLICT DO NOTHING is safe but will not deduplicate by name. If re-running produces duplicates, add a WHERE NOT EXISTS guard on full_name + race_id.
- DO NOT include slug in the INSERT — it is a GENERATED column on essentials.chambers (different table) but verify no generated columns on race_candidates.
