**File structure:**
```typescript
/**
 * smoke-ca-geofences.ts
 * Phase 57: California geofences smoke test
 *
 * Usage:
 *   cd C:/EV-Accounts/backend
 *   npx tsx scripts/smoke-ca-geofences.ts
 *
 * Prints which geofence_boundaries rows cover each test address.
 * Asserts specific mtfcc values per address. Special cases:
 *  - SF City Hall must return BOTH G4110 city AND G4020 county (consolidated)
 *  - East LA must return G4040 (CCD) and must NOT return G4110 (unincorporated)
 *
 * Also queries definitive geo_ids for the 7 v7.0 target cities (SF, LA, San
 * Jose, San Diego, Sacramento, Fremont, Berkeley) so downstream Phases 63-68
 * can wire city governments to deterministic geo_ids.
 */
import { Client } from 'pg';
import * as dotenv from 'dotenv';
dotenv.config();

interface TestAddress {
  label: string;
  lon: number;
  lat: number;
  expectedMtfcc: string[];      // mtfcc values that MUST appear in the result
  forbiddenMtfcc?: string[];    // mtfcc values that MUST NOT appear (East LA case)
  notes: string;
}

const TEST_ADDRESSES: TestAddress[] = [
  {
    label: 'San Francisco City Hall (consolidated city-county)',
    lon: -122.4191,
    lat: 37.7792,
    expectedMtfcc: ['G4020', 'G4110', 'G5200', 'G5210', 'G5220'],
    notes: 'Must return BOTH G4110 (city, geo_id 0667000) AND G4020 (county, geo_id 06075). Expected CD: CA-11 (Pelosi).',
  },
  {
    label: 'San Diego Balboa Park (incorporated, CA-51)',
    lon: -117.1425,
    lat: 32.7308,
    expectedMtfcc: ['G4020', 'G4110', 'G5200', 'G5210', 'G5220'],
    notes: 'Expected G4110: San Diego city (0666000). Expected CD: CA-51 (Jacobs). Expected county: San Diego (06073).',
  },
  {
    label: 'East Los Angeles (unincorporated CCD)',
    lon: -118.1720,
    lat: 34.0239,
    expectedMtfcc: ['G4020', 'G4040', 'G5200', 'G5210', 'G5220'],
    forbiddenMtfcc: ['G4110'],
    notes: 'Must return G4040 CCD. Must NOT return G4110 (East LA is unincorporated). LA County (06037). Expected CD: CA-34 area (Gomez). FALLBACK COORDINATE if this point drifts into City of LA: lon=-118.1629, lat=34.0214.',
  },
];

async function main(): Promise<void> {
  const client = new Client({ connectionString: process.env.DATABASE_URL });
  await client.connect();
  try {
    let failures = 0;

    // ── Per-address point-in-polygon queries ─────────────────────────────────
    for (const addr of TEST_ADDRESSES) {
      console.log(`\n=== ${addr.label} (lon=${addr.lon}, lat=${addr.lat}) ===`);
      console.log(`    ${addr.notes}`);
      const res = await client.query(
        `SELECT mtfcc, geo_id, name
           FROM essentials.geofence_boundaries
          WHERE state = '06'
            AND ST_Covers(geometry, ST_SetSRID(ST_MakePoint($1, $2), 4326))
          ORDER BY mtfcc`,
        [addr.lon, addr.lat],
      );
      const seenMtfcc = new Set(res.rows.map((r) => r.mtfcc as string));
      for (const row of res.rows) {
        console.log(`  ${row.mtfcc}  geo_id=${row.geo_id}  name=${row.name}`);
      }

      // Assertions
      for (const required of addr.expectedMtfcc) {
        if (!seenMtfcc.has(required)) {
          console.error(`  FAIL: expected mtfcc ${required} not returned`);
          failures++;
        }
      }
      for (const forbidden of addr.forbiddenMtfcc ?? []) {
        if (seenMtfcc.has(forbidden)) {
          const offending = res.rows.filter((r) => r.mtfcc === forbidden);
          console.error(`  FAIL: forbidden mtfcc ${forbidden} returned: ${JSON.stringify(offending)}`);
          failures++;
        }
      }
    }

    // ── v7.0 target city geo_id lookup ───────────────────────────────────────
    console.log('\n=== v7.0 Target City G4110 geo_ids (for downstream Phases 63-68) ===');
    const targetCityRes = await client.query(
      `SELECT geo_id, name
         FROM essentials.geofence_boundaries
        WHERE state = '06'
          AND mtfcc = 'G4110'
          AND name IN (
            'San Francisco city',
            'Los Angeles city',
            'San Jose city',
            'San Diego city',
            'Sacramento city',
            'Fremont city',
            'Berkeley city'
          )
        ORDER BY name`,
    );
    for (const row of targetCityRes.rows) {
      console.log(`  ${row.name.padEnd(24)} → geo_id=${row.geo_id}`);
    }
    if (targetCityRes.rows.length !== 7) {
      console.error(`  FAIL: expected 7 target cities, got ${targetCityRes.rows.length}. Missing: ${
        ['San Francisco city','Los Angeles city','San Jose city','San Diego city','Sacramento city','Fremont city','Berkeley city']
          .filter((n) => !targetCityRes.rows.find((r) => r.name === n)).join(', ')
      }`);
      failures++;
    }

    // ── Summary ──────────────────────────────────────────────────────────────
    console.log(`\n${'='.repeat(60)}`);
    if (failures === 0) {
      console.log('Smoke test PASSED. All 3 addresses + 7 target cities verified.');
    } else {
      console.error(`Smoke test FAILED with ${failures} assertion failures.`);
      process.exit(1);
    }
  } finally {
    await client.end();
  }
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```
