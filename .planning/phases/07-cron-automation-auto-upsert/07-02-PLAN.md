   ```typescript
   async function withRetry<T>(fn: () => Promise<T>, label: string): Promise<T> {
     let lastErr: unknown;
     for (let attempt = 0; attempt <= RETRY_DELAYS_MS.length; attempt++) {
       try {
         return await fn();
       } catch (err) {
         lastErr = err;
         if (attempt < RETRY_DELAYS_MS.length && isTransient(err)) {
           const delay = RETRY_DELAYS_MS[attempt];
           console.warn(`[discoveryCron] ${label}: transient error, retrying in ${delay}ms`, err);
           await sleep(delay);
           continue;
         }
         break;
       }
     }
     throw lastErr;
   }
   ```
   Loop runs `attempt = 0..3` (4 total attempts: 1 initial + 3 retries). On non-transient errors, break immediately and rethrow.
