    ```sql
    SELECT
      ct.id AS topic_id,
      ct.topic_key,
      ct.name,
      COALESCE(
        json_agg(
          json_build_object('scope', ctr.role_scope, 'is_required', ctr.is_required)
          ORDER BY ctr.role_scope
        ) FILTER (WHERE ctr.role_scope IS NOT NULL),
        '[]'::json

      'affordable-housing',
      'childcare-affordability',
      'jail-capacity',
      'homelessness',
      'data-center-development'
    ```
