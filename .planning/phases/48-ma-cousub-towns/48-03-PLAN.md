```sql
-- Migration 167: Cambridge offices district_id back-fill
-- 
-- Phase 48 UAT gap: Cambridge address search returns no local officials.
-- Root cause: Cambridge offices have district_id = NULL. getRepresentativesByAddress
-- (essentialsService.ts:579) uses INNER JOIN on district_id, silently dropping
-- all 17 Cambridge offices from address-based results.
--
-- Fix: INSERT one essentials.districts row for Cambridge (geo_id='2511000'),
-- then UPDATE all 17 Cambridge offices (City Council + School Committee chambers)
-- to set district_id to that new UUID.
--
-- Precedent: Migration 115 (LA City Attorney/Controller) applied the identical
-- pattern when offices existed with NULL district_id and address-lookup dropped them.
--
-- Idempotency: essentials.districts has NO unique constraint on geo_id, so we
-- gate the INSERT with WHERE NOT EXISTS. UPDATE is naturally idempotent (sets
-- the same value on re-run).

BEGIN;

-- ─── Step 1: INSERT Cambridge district row (idempotent) ──────────────────────
INSERT INTO essentials.districts (geo_id, district_type, mtfcc, state, label)
SELECT '2511000', 'LOCAL', 'G4110', 'MA', 'Cambridge'
WHERE NOT EXISTS (
  SELECT 1 FROM essentials.districts
  WHERE geo_id = '2511000' AND mtfcc = 'G4110' AND state = 'MA'
);

-- ─── Step 2: UPDATE Cambridge offices to point at new district ────────────────
-- Chambers (from migration 158):
--   City Council:     b4b8c0a1-2658-4df4-9196-9646c99d173c
--   School Committee: 41846a49-e5d5-460d-b2c2-0f4f8130b949
-- Government: Cambridge geo_id='2511000' (UUID 6f7d55bc-d50c-47ff-b521-5767d1f763fb)

UPDATE essentials.offices
SET district_id = (
  SELECT id FROM essentials.districts
  WHERE geo_id = '2511000' AND mtfcc = 'G4110' AND state = 'MA'
  LIMIT 1
)
WHERE chamber_id IN (
  'b4b8c0a1-2658-4df4-9196-9646c99d173c',  -- Cambridge City Council
  '41846a49-e5d5-460d-b2c2-0f4f8130b949'   -- Cambridge School Committee
)
AND district_id IS NULL;

-- ─── Step 3: Verification gates (raise exception on failure) ─────────────────
DO $$
DECLARE
  v_district_count INT;
  v_offices_with_district INT;
  v_offices_total INT;
BEGIN
  -- Gate 1: exactly one Cambridge district row exists
  SELECT COUNT(*) INTO v_district_count
  FROM essentials.districts
  WHERE geo_id = '2511000' AND mtfcc = 'G4110' AND state = 'MA';

  IF v_district_count <> 1 THEN
    RAISE EXCEPTION 'Migration 167 FAILED: expected 1 Cambridge district row, found %', v_district_count;
  END IF;

  -- Gate 2: all 17 Cambridge offices now have district_id set
  SELECT COUNT(*) INTO v_offices_total
  FROM essentials.offices
  WHERE chamber_id IN (
    'b4b8c0a1-2658-4df4-9196-9646c99d173c',
    '41846a49-e5d5-460d-b2c2-0f4f8130b949'
  );

  SELECT COUNT(*) INTO v_offices_with_district
  FROM essentials.offices
  WHERE chamber_id IN (
    'b4b8c0a1-2658-4df4-9196-9646c99d173c',
    '41846a49-e5d5-460d-b2c2-0f4f8130b949'
  )
  AND district_id IS NOT NULL;

  IF v_offices_total <> 17 THEN
    RAISE EXCEPTION 'Migration 167 FAILED: expected 17 Cambridge offices, found %', v_offices_total;
  END IF;

  IF v_offices_with_district <> 17 THEN
    RAISE EXCEPTION 'Migration 167 FAILED: expected 17 Cambridge offices with district_id, found %', v_offices_with_district;
  END IF;

  RAISE NOTICE 'Migration 167 OK: 1 Cambridge district, % offices linked', v_offices_with_district;
END $$;

COMMIT;
```

Key correctness notes (apply during writing):

- Use `WHERE NOT EXISTS` because `essentials.districts` has NO unique constraint on geo_id — `ON CONFLICT` would fail with "no unique or exclusion constraint matching the ON CONFLICT specification".
- Wrap entire migration in `BEGIN; ... COMMIT;` so the verification gates' `RAISE EXCEPTION` rolls back the whole migration on failure.
- The UPDATE filters `AND district_id IS NULL` so re-running won't clobber any legitimate non-NULL value (defensive — shouldn't matter today, future-proof).
- Do NOT touch `essentials.governments` — the Cambridge government row already exists from prior phases.
- Do NOT modify `essentialsService.ts` — application code is correct; the data is the bug.
