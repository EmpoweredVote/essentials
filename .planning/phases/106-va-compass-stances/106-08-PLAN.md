      SELECT p.external_id, p.full_name, COUNT(pa.*) AS stance_count,
             CASE
               WHEN p.external_id = -510001 AND COUNT(pa.*) >= 15 THEN 'OK'
               WHEN p.external_id IN (-510002, -510003) AND COUNT(pa.*) >= 10 THEN 'OK'
               WHEN p.external_id IN (-400079, -400080) AND COUNT(pa.*) >= 15 THEN 'OK'
               ELSE 'BELOW_MINIMUM'
             END AS status
      FROM essentials.politicians p
      LEFT JOIN inform.politician_answers pa ON pa.politician_id = p.id
      WHERE p.external_id IN (-510001, -510002, -510003, -400079, -400080)
      GROUP BY p.external_id, p.full_name
      ORDER BY p.external_id DESC;
    Required: 5 rows ALL with status = 'OK'. Any BELOW_MINIMUM is a phase failure — escalate by returning a gap-closure recommendation to the orchestrator.

    Q2 — Phase-wide 100% citation rate (D-10) across ALL phase-106 politicians:
      SELECT COUNT(*) AS uncited_total
      FROM inform.politician_context pc
      JOIN essentials.politicians p ON p.id = pc.politician_id
      WHERE (p.external_id IN (-510001, -510002, -510003, -400079, -400080)
             OR p.external_id BETWEEN -5101000007 AND -5101000001
             OR p.external_id BETWEEN -5100090009 AND -5100090001)
        AND (pc.sources IS NULL OR array_length(pc.sources, 1) IS NULL OR array_length(pc.sources, 1) = 0);
    Required: 0.

    Q3 — Phase-wide pairing check (every answer has a context row):
      SELECT COUNT(*) AS unpaired_total
      FROM inform.politician_answers pa
      JOIN essentials.politicians p ON p.id = pa.politician_id
      LEFT JOIN inform.politician_context pc
        ON pc.politician_id = pa.politician_id AND pc.topic_id = pa.topic_id
      WHERE (p.external_id IN (-510001, -510002, -510003, -400079, -400080)
             OR p.external_id BETWEEN -5101000007 AND -5101000001
             OR p.external_id BETWEEN -5100090009 AND -5100090001)
        AND pc.politician_id IS NULL;
    Required: 0.

    Q4 — VA-STANCES-03 coverage stats (informational, not a gate):
      SELECT
        SUM(CASE WHEN external_id BETWEEN -5101000007 AND -5101000001 AND stance_count > 0 THEN 1 ELSE 0 END) AS council_with_stances,
        SUM(CASE WHEN external_id BETWEEN -5101000007 AND -5101000001 THEN 1 ELSE 0 END) AS council_total,
        SUM(CASE WHEN external_id BETWEEN -5100090009 AND -5100090001 AND stance_count > 0 THEN 1 ELSE 0 END) AS acps_with_stances,
        SUM(CASE WHEN external_id BETWEEN -5100090009 AND -5100090001 THEN 1 ELSE 0 END) AS acps_total
      FROM (
        SELECT p.external_id, COUNT(pa.*) AS stance_count
        FROM essentials.politicians p
        LEFT JOIN inform.politician_answers pa ON pa.politician_id = p.id
        WHERE p.external_id BETWEEN -5101000007 AND -5101000001
           OR p.external_id BETWEEN -5100090009 AND -5100090001
        GROUP BY p.external_id
      ) t;
    Required: council_total=7, acps_total=9 (always); with_stances counts captured for SUMMARY (best-effort).
