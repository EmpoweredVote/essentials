```sql
-- ===== {geo_id}: {Name} ({Party initial}) =====
WITH ins_p AS (
  INSERT INTO essentials.politicians
    (id, full_name, first_name, last_name, party, is_active, is_appointed,
     is_vacant, is_incumbent, external_id, urls)
  VALUES (gen_random_uuid(), '{Full Name}', '{First}', '{Last}', '{Party}',
          true, false, false, true, {external_id},
          ARRAY['https://malegislature.gov/Legislators/Profile/{code}'])
  ON CONFLICT (external_id) DO NOTHING
  RETURNING id
)
INSERT INTO essentials.offices
  (id, district_id, chamber_id, politician_id, title, representing_state,
   is_appointed_position, is_vacant)
SELECT gen_random_uuid(),
       d.id,
       (SELECT id FROM essentials.chambers WHERE name = 'Massachusetts Senate'),
       p.id,
       'Senator, {DB District Label}', 'MA', false, false
FROM essentials.districts d
CROSS JOIN ins_p p
WHERE d.geo_id = '{geo_id}' AND d.district_type = 'STATE_UPPER' AND d.state = 'ma'
  AND NOT EXISTS (
    SELECT 1 FROM essentials.offices o
    WHERE o.district_id = d.id
      AND o.chamber_id = (SELECT id FROM essentials.chambers WHERE name = 'Massachusetts Senate')
  );
```
