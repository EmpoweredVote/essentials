```typescript
/**
 * load-collin-county-boundary.ts
 *
 * Downloads the TIGER/Line 2024 US Counties shapefile, finds the Collin County, TX
 * record (FIPS 48085 — STATEFP=48, COUNTYFP=085), and inserts its geometry into
 * essentials.geofence_boundaries with mtfcc='G4020'.
 *
 * Idempotent: ON CONFLICT (geo_id, mtfcc) DO NOTHING.
 *
 * This is intentionally scoped to a single county. We could later generalize to
 * load all TX counties, but for Phase 19 we only need Collin County to enable
 * the by-government-list congressional intersection query.
 *
 * Run: cd /c/EV-Accounts/backend && npx tsx scripts/load-collin-county-boundary.ts
 */

import { Pool } from 'pg';
import * as path from 'path';
import * as fs from 'fs';
import * as os from 'os';
import * as https from 'https';
import AdmZip from 'adm-zip';
import * as shapefile from 'shapefile';
import * as dotenv from 'dotenv';

dotenv.config({ path: path.join(__dirname, '..', '.env') });

const TIGER_URL = 'https://www2.census.gov/geo/tiger/TIGER2024/COUNTY/tl_2024_us_county.zip';
const TARGET_STATEFP = '48';     // Texas
const TARGET_COUNTYFP = '085';   // Collin
const TARGET_GEOID = '48085';
const TARGET_MTFCC = 'G4020';

async function downloadFile(url: string, dest: string): Promise<void> {
  return new Promise((resolve, reject) => {
    const file = fs.createWriteStream(dest);
    https.get(url, (response) => {
      // Handle redirects
      if (response.statusCode === 301 || response.statusCode === 302) {
        const redirectUrl = response.headers.location;
        if (!redirectUrl) return reject(new Error('Redirect with no location'));
        file.close();
        fs.unlinkSync(dest);
        return downloadFile(redirectUrl, dest).then(resolve).catch(reject);
      }
      if (response.statusCode !== 200) {
        return reject(new Error(`HTTP ${response.statusCode}`));
      }
      response.pipe(file);
      file.on('finish', () => file.close(() => resolve()));
    }).on('error', (err) => {
      fs.unlink(dest, () => reject(err));
    });
  });
}

async function main() {
  const databaseUrl = process.env.DATABASE_URL;
  if (!databaseUrl) {
    throw new Error('DATABASE_URL not set — source backend/.env first.');
  }

  const tmpDir = fs.mkdtempSync(path.join(os.tmpdir(), 'collin-county-'));
  const zipPath = path.join(tmpDir, 'tl_2024_us_county.zip');

  console.log(`Downloading ${TIGER_URL}...`);
  await downloadFile(TIGER_URL, zipPath);
  console.log(`Downloaded to ${zipPath} (${(fs.statSync(zipPath).size / 1024 / 1024).toFixed(1)}MB)`);

  console.log('Extracting...');
  const zip = new AdmZip(zipPath);
  zip.extractAllTo(tmpDir, true);

  // Find the .shp file
  const files = fs.readdirSync(tmpDir);
  const shpFile = files.find((f) => f.endsWith('.shp'));
  if (!shpFile) throw new Error('No .shp file found in extracted ZIP');
  const shpPath = path.join(tmpDir, shpFile);

  console.log(`Reading shapefile ${shpFile}...`);
  const source = await shapefile.open(shpPath);

  let collinFeature: { properties: any; geometry: any } | null = null;
  let totalCounties = 0;

  while (true) {
    const result = await source.read();
    if (result.done) break;
    totalCounties++;
    const props = result.value.properties as any;
    if (props.STATEFP === TARGET_STATEFP && props.COUNTYFP === TARGET_COUNTYFP) {
      collinFeature = result.value as any;
      break;
    }
  }

  console.log(`Scanned ${totalCounties} county records...`);
  if (!collinFeature) {
    throw new Error(`Collin County (STATEFP=${TARGET_STATEFP}, COUNTYFP=${TARGET_COUNTYFP}) not found.`);
  }

  const props = collinFeature.properties;
  console.log(`Found Collin County: GEOID=${props.GEOID}, NAMELSAD=${props.NAMELSAD}, MTFCC=${props.MTFCC}`);

  // Confirm GEOID and MTFCC match expectations (research said G4020 — verify in shapefile)
  if (props.GEOID !== TARGET_GEOID) {
    throw new Error(`Unexpected GEOID: got ${props.GEOID}, expected ${TARGET_GEOID}`);
  }

  // Some county shapefiles use G4020 (county subdivision) — but the TIGER county
  // shapefile actually uses MTFCC G4020 for "County or Equivalent Feature".
  // If shapefile MTFCC differs from G4020, we still write G4020 because that's
  // what essentialsBrowseService.ts expects. Log a warning if they differ.
  const shapefileMtfcc = props.MTFCC;
  if (shapefileMtfcc !== TARGET_MTFCC) {
    console.warn(`WARNING: shapefile MTFCC=${shapefileMtfcc}, but inserting as ${TARGET_MTFCC} (matches getAreasForState query).`);
  }

  const ocdId = `ocd-division/country:us/state:tx/county:collin`;
  const name = props.NAMELSAD || 'Collin County';
  const state = TARGET_STATEFP; // '48' — matches existing convention
  const geoJson = JSON.stringify(collinFeature.geometry);

  const pool = new Pool({ connectionString: databaseUrl });
  try {
    const result = await pool.query(`
      INSERT INTO essentials.geofence_boundaries
        (geo_id, ocd_id, name, state, mtfcc, geometry, source, imported_at)
      VALUES (
        $1, $2, $3, $4, $5,
        public.ST_SetSRID(public.ST_Force2D(public.ST_GeomFromGeoJSON($6)), 4326),
        'census_tiger_2024',
        now()
      )
      ON CONFLICT (geo_id, mtfcc) DO NOTHING
      RETURNING geo_id;
    `, [TARGET_GEOID, ocdId, name, state, TARGET_MTFCC, geoJson]);

    if (result.rowCount && result.rowCount > 0) {
      console.log(`Inserted Collin County boundary (geo_id=${TARGET_GEOID}, mtfcc=${TARGET_MTFCC}).`);
    } else {
      console.log(`Collin County boundary already exists (geo_id=${TARGET_GEOID}, mtfcc=${TARGET_MTFCC}). Skipped.`);
    }
  } finally {
    await pool.end();
    // Clean up tmp files
    fs.rmSync(tmpDir, { recursive: true, force: true });
    console.log('Cleaned up temp files.');
  }
}

main().catch((err) => {
  console.error('FATAL:', err);
  process.exit(1);
});
```

Critical rules:
- The script MUST use the same `ST_SetSRID(ST_Force2D(ST_GeomFromGeoJSON(...)), 4326)` pattern as the congressional loader.
- The `state` field MUST be `'48'` (2-digit FIPS), NOT `'TX'` — research is explicit about this.
- The mtfcc inserted MUST be `'G4020'` even if the shapefile property differs — `essentialsBrowseService.ts` queries on `mtfcc='G4020'`.
- ON CONFLICT (geo_id, mtfcc) DO NOTHING for idempotency.
- No hardcoded credentials — load from `.env` via dotenv.
- Clean up the tmp directory at the end (TIGER county zip is ~120MB).
