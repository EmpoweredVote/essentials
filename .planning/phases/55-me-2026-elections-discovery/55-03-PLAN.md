Run the following verification queries against the live DB and collect results:

```sql
-- 1. Election rows
SELECT name, election_date, election_type, jurisdiction_level, state
FROM essentials.elections
WHERE state = 'ME'
ORDER BY election_date;

-- 2. Total race count by type
SELECT
  CASE
    WHEN r.position_name LIKE 'ME State Senate%' THEN 'Senate Legislative'
    WHEN r.position_name LIKE 'ME State House%' THEN 'House Legislative'
    ELSE 'Statewide/Federal'
  END AS race_type,
  e.election_date,
  COUNT(*) AS race_count
FROM essentials.races r
JOIN essentials.elections e ON e.id = r.election_id
WHERE e.state = 'ME' AND e.election_date < '2027-01-01'
GROUP BY 1, 2
ORDER BY 2, 1;
-- Expected:
--  Senate Legislative  | 2026-06-09 | 35
--  Senate Legislative  | 2026-11-03 | 35
--  House Legislative   | 2026-06-09 | 151
--  House Legislative   | 2026-11-03 | 151
--  Statewide/Federal   | 2026-06-09 | 4 (Gov, Senate, ME-01, ME-02 primaries)
--  Statewide/Federal   | 2026-11-03 | 4 (Gov, Senate, ME-01, ME-02 generals)

-- 3. Statewide candidate list with is_incumbent flag
SELECT r.position_name, e.election_type,
       rc.full_name, rc.is_incumbent,
       CASE WHEN rc.politician_id IS NOT NULL THEN 'linked' ELSE 'unlinked' END AS politician_link
FROM essentials.race_candidates rc
JOIN essentials.races r ON r.id = rc.race_id
JOIN essentials.elections e ON e.id = r.election_id
WHERE e.state = 'ME' AND e.election_date < '2027-01-01'
  AND r.position_name NOT LIKE 'ME State%'
ORDER BY r.position_name, e.election_type, rc.full_name;
-- Key checks:
--   Janet Mills should NOT appear
--   Jared Golden should NOT appear as is_incumbent=true (or at all)
--   Chellie Pingree should appear with is_incumbent=true on ME-01 primary
--   Susan Collins should appear with is_incumbent=true on U.S. Senate Maine primary
--   Graham Platner should appear on U.S. Senate Maine primary
--   David Costello should appear on U.S. Senate Maine primary

-- 4. Discovery jurisdictions + cron scope check
SELECT jurisdiction_geoid, jurisdiction_name, election_date,
       election_date - CURRENT_DATE AS days_until,
       CASE WHEN election_date - CURRENT_DATE <= 180 THEN 'IN SCOPE' ELSE 'OUT OF SCOPE' END AS cron_status
FROM essentials.discovery_jurisdictions
WHERE state = 'ME'
ORDER BY election_date;
-- Expected:
--   '23' | State of Maine | 2026-06-09 | ~20 days | IN SCOPE
--   '23' | State of Maine | 2026-11-03 | ~166 days | IN SCOPE
--   '2360545' | City of Portland, Maine | 2027-11-02 | ~532 days | OUT OF SCOPE

-- 5. office_id coverage on legislative races
SELECT
  CASE WHEN r.position_name LIKE '%Senate%' THEN 'Senate' ELSE 'House' END AS chamber,
  COUNT(*) AS total_races,
  COUNT(r.office_id) AS with_office_id,
  COUNT(*) - COUNT(r.office_id) AS missing_office_id
FROM essentials.races r
JOIN essentials.elections e ON e.id = r.election_id
WHERE e.state = 'ME' AND r.position_name LIKE 'ME State%'
GROUP BY 1;
-- Expected: Senate 70 total / 70 with office_id / 0 missing
--           House  302 total / ~301 with / ~1 missing (D29 vacant seat may have NULL)
```
