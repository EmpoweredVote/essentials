       p.id,
       'U.S. Representative', 'CA', false, false, NULL
FROM essentials.districts d
CROSS JOIN ins_p p
WHERE d.geo_id = '{geo_id}' AND d.district_type = 'NATIONAL_LOWER' AND d.state = 'CA'

  );
