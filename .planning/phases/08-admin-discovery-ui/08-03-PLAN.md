```javascript
/**
 * Fetch all discovery jurisdictions with last-run summary and active candidate count.
 * @returns {Promise<{jurisdictions: Array}>}
 */
export async function fetchDiscoveryJurisdictions() {
  const res = await apiFetch('/admin/discovery/jurisdictions');
  if (!res) { const err = new Error('Unauthorized'); err.status = 401; throw err; }
  if (!res.ok) { const err = new Error(`Fetch jurisdictions failed: ${res.status}`); err.status = res.status; throw err; }
  return res.json();
}

/**
 * Fetch paginated discovery run history.
 * @param {Object} params
 * @param {number} [params.limit=25]
 * @param {number} [params.offset=0]
 * @param {string} [params.jurisdiction_id] - UUID filter
 * @returns {Promise<{runs: Array, total: number, limit: number, offset: number}>}
 */
export async function fetchDiscoveryRuns({ limit = 25, offset = 0, jurisdiction_id } = {}) {
  const qs = new URLSearchParams();
  qs.set('limit', String(limit));
  qs.set('offset', String(offset));
  if (jurisdiction_id) qs.set('jurisdiction_id', jurisdiction_id);
  const res = await apiFetch(`/admin/discovery/runs?${qs}`);
  if (!res) { const err = new Error('Unauthorized'); err.status = 401; throw err; }
  if (!res.ok) { const err = new Error(`Fetch runs failed: ${res.status}`); err.status = res.status; throw err; }
  return res.json();
}

/**
 * Fetch per-jurisdiction coverage health stats.
 * @returns {Promise<{coverage: Array}>}
 */
export async function fetchDiscoveryCoverage() {
  const res = await apiFetch('/admin/discovery/coverage');
  if (!res) { const err = new Error('Unauthorized'); err.status = 401; throw err; }
  if (!res.ok) { const err = new Error(`Fetch coverage failed: ${res.status}`); err.status = res.status; throw err; }
  return res.json();
}

/**
 * Trigger a discovery run for a specific jurisdiction.
 * Returns the parsed response on 202; throws on non-2xx (caller should detect 409 specifically).
 * @param {string} jurisdictionId
 * @returns {Promise<Response>} raw response so caller can branch on 409
 */
export async function triggerDiscoveryRun(jurisdictionId) {
  const res = await apiFetch(`/admin/discover/jurisdiction/${encodeURIComponent(jurisdictionId)}`, { method: 'POST' });
  if (!res) { const err = new Error('Unauthorized'); err.status = 401; throw err; }
  return res; // caller inspects res.status (202 success, 409 lock conflict, others=error)
}
```

```jsx
  fetchDiscoveryJurisdictions,
  fetchDiscoveryRuns,
  fetchDiscoveryCoverage,

} from '../../lib/adminApi';

  success: 'Success',
  failed: 'Failed',
  running: 'Running',
  never_run: 'Never run',
};
const POLL_INTERVAL_MS = 4000;
const RUNS_PER_PAGE = 25;

function formatDateTime(s) { /* en-US locale month+day+year+time */ }
function formatDate(s)     { /* en-US locale month+day+year */ }
function statusKey(row)    { return row.last_run_status ?? 'never_run'; }
function triggerLabel(t)   { return t === 'cron' ? 'Cron' : 'Manual'; }

function StatusBadge({ status }) { /* colored pill from STATUS_STYLES */ }
function Spinner() { /* the standard ev-teal animate-spin border ring */ }

function Toast({ toast, onClose }) { /* fixed top-right, 5s auto-close, green/red variants */ }

export default function DiscoveryDashboard() {
  // Section 1: Jurisdictions
  const [jurisdictions, setJurisdictions] = useState([]);
  const [jLoading, setJLoading] = useState(true);
  const [search, setSearch] = useState('');
  const [sortKey, setSortKey] = useState('election_date');
  const [sortDir, setSortDir] = useState('asc');

  // Section 2: Runs history
  const [runs, setRuns] = useState([]);
  const [runsTotal, setRunsTotal] = useState(0);
  const [runsPage, setRunsPage] = useState(0);
  const [runsFilterJurisdiction, setRunsFilterJurisdiction] = useState('');

  // Section 3: Coverage
  const [coverage, setCoverage] = useState([]);

  const [toast, setToast] = useState(null);

  const loadJurisdictions = useCallback(async () => {
    try {
      const data = await fetchDiscoveryJurisdictions();
      setJurisdictions(data.jurisdictions || []);
    } catch (err) { setToast({ type: 'error', message: err.message }); }
    finally { setJLoading(false); }
  }, []);

  const loadRuns = useCallback(async () => {
    try {
      const data = await fetchDiscoveryRuns({
        limit: RUNS_PER_PAGE,
        offset: runsPage * RUNS_PER_PAGE,
        jurisdiction_id: runsFilterJurisdiction || undefined,
      });
      setRuns(data.runs || []);
      setRunsTotal(data.total || 0);
    } catch (err) { setToast({ type: 'error', message: err.message }); }
  }, [runsPage, runsFilterJurisdiction]);

  const loadCoverage = useCallback(async () => {
    try {
      const data = await fetchDiscoveryCoverage();
      setCoverage(data.coverage || []);
    } catch (err) { setToast({ type: 'error', message: err.message }); }
  }, []);

  // Initial load
  useEffect(() => { loadJurisdictions(); loadCoverage(); }, [loadJurisdictions, loadCoverage]);
  useEffect(() => { loadRuns(); }, [loadRuns]);

  // Poll while any jurisdiction is running. Capture previous statuses to detect transitions
  // and refresh runs+coverage exactly when a row transitions running -> settled.
  useEffect(() => {
    const isAnyRunning = jurisdictions.some(j => j.last_run_status === 'running');
    if (!isAnyRunning) return;
    const id = setInterval(async () => {
      const prev = jurisdictions;
      try {
        const data = await fetchDiscoveryJurisdictions();
        const next = data.jurisdictions || [];
        setJurisdictions(next);
        // Detect transitions
        for (const n of next) {
          const p = prev.find(x => x.id === n.id);
          if (p?.last_run_status === 'running' && n.last_run_status !== 'running') {
            // Settled -- refresh other sections and toast
            loadRuns();
            loadCoverage();
            if (n.last_run_status === 'success') {
              setToast({ type: 'success', message: `${n.jurisdiction_name}: discovery complete (${n.last_run_candidates_added ?? 0} staged).` });
            } else if (n.last_run_status === 'failed') {
              setToast({ type: 'error', message: `${n.jurisdiction_name}: discovery failed.` });
            }
          }
        }
      } catch { /* swallow during polling */ }
    }, POLL_INTERVAL_MS);
    return () => clearInterval(id);
  }, [jurisdictions, loadRuns, loadCoverage]);

  // Sorting + searching
  const visibleJurisdictions = useMemo(() => {
    const q = search.trim().toLowerCase();
    let rows = jurisdictions.filter(j =>
      !q || (j.jurisdiction_name || '').toLowerCase().includes(q)
    );
    rows = [...rows].sort((a, b) => {
      const A = a[sortKey] ?? '';
      const B = b[sortKey] ?? '';
      const cmp = A < B ? -1 : A > B ? 1 : 0;
      return sortDir === 'asc' ? cmp : -cmp;
    });
    return rows;
  }, [jurisdictions, search, sortKey, sortDir]);

  function toggleSort(key) {
    if (sortKey === key) setSortDir(d => d === 'asc' ? 'desc' : 'asc');
    else { setSortKey(key); setSortDir('asc'); }
  }

  async function handleRunDiscovery(j) {
    // Optimistic update -- prevents double-click 409 races (research Pitfall 6)

    try {

      if (res.status === 409) {
        const body = await res.json().catch(() => ({}));
        setToast({ type: 'error', message: body.message || 'Run already in progress' });
        loadJurisdictions(); // resync truth
        return;
      }
      if (!res.ok && res.status !== 202) {
        setToast({ type: 'error', message: `Trigger failed: ${res.status}` });
        loadJurisdictions();
        return;
      }
      // 202 accepted -- polling effect handles the rest.
    } catch (err) {
      setToast({ type: 'error', message: err.message });
      loadJurisdictions();
    }
  }

  return (
    <div className="min-h-screen bg-gray-50 p-6">
      <Toast toast={toast} onClose={() => setToast(null)} />
      <div className="max-w-7xl mx-auto space-y-10">

        <header>
          <h1 className="text-2xl font-bold">Discovery Dashboard</h1>
          <p className="text-sm text-gray-600">Trigger and monitor candidate discovery runs across jurisdictions.</p>
        </header>

        {/* SECTION 1: JURISDICTIONS TABLE */}
        <section>
          <div className="flex items-center justify-between mb-3">
            <h2 className="text-lg font-semibold">Jurisdictions</h2>
            <input
              type="text" value={search} onChange={e => setSearch(e.target.value)}
              placeholder="Search jurisdictions..."
              className="border rounded px-3 py-1 text-sm w-64"
            />
          </div>
          <div className="bg-white rounded shadow overflow-hidden">
            <table className="w-full text-sm">
              <thead className="bg-gray-100 text-left">
                <tr>
                  <Th label="Jurisdiction" k="jurisdiction_name" sortKey={sortKey} sortDir={sortDir} onClick={toggleSort} />
                  <Th label="Election" k="election_date" sortKey={sortKey} sortDir={sortDir} onClick={toggleSort} />
                  <Th label="Status" k="last_run_status" sortKey={sortKey} sortDir={sortDir} onClick={toggleSort} />
                  <Th label="Last Run" k="last_run_started_at" sortKey={sortKey} sortDir={sortDir} onClick={toggleSort} />
                  <Th label="Last Added" k="last_run_candidates_added" sortKey={sortKey} sortDir={sortDir} onClick={toggleSort} />
                  <Th label="Total Active" k="total_active_candidates" sortKey={sortKey} sortDir={sortDir} onClick={toggleSort} />
                  <th className="px-3 py-2"></th>
                </tr>
              </thead>
              <tbody>
                {jLoading && <tr><td colSpan={7} className="px-3 py-4 text-center text-gray-500">Loading...</td></tr>}
                {!jLoading && visibleJurisdictions.length === 0 && <tr><td colSpan={7} className="px-3 py-4 text-center text-gray-500">No jurisdictions match.</td></tr>}
                {visibleJurisdictions.map(j => (
                  <tr key={j.id} className="border-t">
                    <td className="px-3 py-2 font-medium">{j.jurisdiction_name}</td>
                    <td className="px-3 py-2">{formatDate(j.election_date)}</td>
                    <td className="px-3 py-2"><StatusBadge status={statusKey(j)} /></td>
                    <td className="px-3 py-2 text-gray-600">{formatDateTime(j.last_run_started_at)}</td>
                    <td className="px-3 py-2">{j.last_run_candidates_added ?? '—'}</td>
                    <td className="px-3 py-2">{j.total_active_candidates ?? 0}</td>
                    <td className="px-3 py-2 text-right">
                      {j.last_run_status === 'running'
                        ? <Spinner />
                        : <button
                            onClick={() => handleRunDiscovery(j)}
                            className="bg-[var(--ev-teal)] text-white text-xs px-3 py-1 rounded hover:opacity-90"
                          >Run Discovery</button>}
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>

        {/* SECTION 2: RUN HISTORY */}
        <section>

            <select

              onChange={e => { setRunsFilterJurisdiction(e.target.value); setRunsPage(0); }}

          <div className="bg-white rounded shadow overflow-hidden">
            <table className="w-full text-sm">

                  <th className="px-3 py-2">Jurisdiction</th>
                  <th className="px-3 py-2">When</th>
                  <th className="px-3 py-2">Found</th>
                  <th className="px-3 py-2">Staged</th>
```
