```sql
-- Migration 105: TX US House Representatives (119th Congress)
--
-- Seeds 37 TX House politicians + 38 offices (TX-1..TX-38; TX-23 is vacant
-- as of April 14, 2026 — office row exists with is_vacant=true, no politician).
--
-- Depends on Plan 19-01 having loaded the 38 NATIONAL_LOWER TX districts
-- (geo_ids '4801'..'4838') via load-us-congressional-boundaries.ts.
--
-- Chamber UUID resolved from live DB: <UUID_FROM_TASK_1>
-- (chamber.name_formal = 'United States House of Representatives')

BEGIN;

-- Helper CTE: get the US House chamber id once (hardcoded for clarity, matching
-- migration 103's pattern of hardcoding the Senate UUID).
-- chamber_id = '<UUID_FROM_TASK_1>'

-- TX-1: Nathaniel Moran (R)
WITH ins_p AS (
  INSERT INTO essentials.politicians
    (id, full_name, first_name, last_name, party, is_active, is_appointed, is_vacant, is_incumbent, external_id)
  VALUES (gen_random_uuid(), 'Nathaniel Moran', 'Nathaniel', 'Moran', 'Republican', true, false, false, true, -100301)
  ON CONFLICT (external_id) DO NOTHING
  RETURNING id
)
INSERT INTO essentials.offices
  (id, district_id, chamber_id, politician_id, title, representing_state, is_appointed_position, is_vacant)
SELECT gen_random_uuid(), d.id, '<UUID_FROM_TASK_1>'::uuid, p.id,
       'Representative', 'TX', false, false
FROM essentials.districts d
CROSS JOIN ins_p p
WHERE d.geo_id = '4801' AND d.district_type = 'NATIONAL_LOWER'
  AND NOT EXISTS (
    SELECT 1 FROM essentials.offices o
    WHERE o.district_id = d.id AND o.chamber_id = '<UUID_FROM_TASK_1>'::uuid
  );

-- TX-2: Dan Crenshaw (R)
WITH ins_p AS (
  INSERT INTO essentials.politicians
    (id, full_name, first_name, last_name, party, is_active, is_appointed, is_vacant, is_incumbent, external_id)
  VALUES (gen_random_uuid(), 'Dan Crenshaw', 'Dan', 'Crenshaw', 'Republican', true, false, false, true, -100302)
  ON CONFLICT (external_id) DO NOTHING
  RETURNING id
)
INSERT INTO essentials.offices
  (id, district_id, chamber_id, politician_id, title, representing_state, is_appointed_position, is_vacant)
SELECT gen_random_uuid(), d.id, '<UUID_FROM_TASK_1>'::uuid, p.id,
       'Representative', 'TX', false, false

CROSS JOIN ins_p p
WHERE d.geo_id = '4802' AND d.district_type = 'NATIONAL_LOWER'
  AND NOT EXISTS (
    SELECT 1 FROM essentials.offices o
    WHERE o.district_id = d.id AND o.chamber_id = '<UUID_FROM_TASK_1>'::uuid
  );

-- ... [Repeat the same CTE pattern for TX-3 through TX-22, then SKIP TX-23,
--      then TX-24 through TX-38. 37 politician INSERTs total.]

COMMIT;
```

Critical rules:
- Replace EVERY `<UUID_FROM_TASK_1>` with the actual UUID captured in Task 1. There will be ~76 occurrences (one per TX-N block × 2 references per block).
- The SQL is wrapped in a single BEGIN/COMMIT transaction so a partial failure rolls back cleanly.
- Each politician INSERT uses `ON CONFLICT (external_id) DO NOTHING` — idempotent on re-runs.
- Each office INSERT uses `NOT EXISTS (... WHERE district_id = d.id AND chamber_id = ...)` — idempotent on re-runs.
- The CROSS JOIN with `ins_p p` resolves to zero rows if the politician INSERT was skipped (already exists), so the office INSERT is also a no-op on a second run. This is the correct idempotency behavior — the FIRST run creates both, subsequent runs do nothing.
- TX-23 has its own standalone INSERT (no politician CTE) with `is_vacant = true` and `politician_id = NULL`.
- Politician fields: `is_active=true`, `is_appointed=false`, `is_vacant=false`, `is_incumbent=true`. Office fields: `is_appointed_position=false`, `is_vacant=false` (except TX-23 = true), `representing_state='TX'`, `title='Representative'`.
- DO NOT include `slug` anywhere — it is a GENERATED column on essentials.chambers (and the same rule of caution applies generally; politicians/offices/districts do not have slug on insert).
- Parties use full strings 'Republican' and 'Democrat' (matching migration 103's convention). DO NOT use 'R' / 'D' / 'GOP' / 'Dem'.
- Use `LF` line endings; UTF-8 encoding (no BOM).
