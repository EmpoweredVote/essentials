   ```ts
   /**
    * Canonical name normalization for fuzzy matching.
    * Lowercase, collapse whitespace, strip common suffixes (Jr, Sr, II, III, IV, V, Esq, PhD, MD).
    * Punctuation (periods, commas, apostrophes) is removed so "O'Brien" and "OBrien" compare equal.
    */
   export function normalizeName(name: string): string {
     return name
       .toLowerCase()
       .replace(/[.,'`’]/g, '')              // strip punctuation incl. smart apostrophe
       .replace(/\s+/g, ' ')
       .trim()
       .replace(
         /\s+(jr|sr|ii|iii|iv|v|esq|phd|md)\s*$/i,
         ''
       )
       .trim();
   }

   /**
    * Similarity in [0, 1] between two names after normalization.
    * 1.0 = identical; 0.85+ is the project-wide "same person" threshold (per 05-CONTEXT).
    * Uses Levenshtein distance: 1 - distance / max(len).
    */
   export function nameSimilarity(a: string, b: string): number {
     const na = normalizeName(a);
     const nb = normalizeName(b);
     const maxLen = Math.max(na.length, nb.length);
     if (maxLen === 0) return 1.0;
     return 1 - distance(na, nb) / maxLen;
   }

   export const NAME_MATCH_THRESHOLD = 0.85;

   /**
    * Score a discovered candidate's confidence from two signals:
    *   1. domainOnAllowlist — is the citation URL's domain on this jurisdiction's allowed_domains?
    *   2. nameMatchesExistingCandidate — does the discovered name fuzzy-match an existing race_candidate at >= threshold?
    *
    * Rules (per 05-CONTEXT):
    *   - official  : domainOnAllowlist is true (name match supports but is NOT required)
    *   - matched   : domainOnAllowlist false, name match >= 0.85
    *   - uncertain : neither signal
    *
    * When allowedDomains is empty/null, domainOnAllowlist is always false — ceiling is 'matched' or 'uncertain'.
    */
   export function scoreConfidence(args: {
     domainOnAllowlist: boolean;
     nameMatchesExistingCandidate: boolean;
   }): 'official' | 'matched' | 'uncertain' {
     if (args.domainOnAllowlist) return 'official';
     if (args.nameMatchesExistingCandidate) return 'matched';
     return 'uncertain';
   }

   /**
    * Extract the registrable domain portion of a URL for allowlist comparison.
    * Strips protocol, port, path, query, fragment, and optional leading "www.".
    * Returns lowercased hostname, e.g. "lavote.gov" for "https://www.lavote.gov/home/voting-elections".
    * Returns null for URLs that can't be parsed.
    */
   export function hostFromUrl(url: string): string | null {
     try {
       const u = new URL(url);
       return u.hostname.toLowerCase().replace(/^www\./, '');
     } catch {
       return null;
     }
   }

   /**
    * Is the URL's host on the allowlist? Matches the domain itself or any subdomain
    * (e.g. "registrar.lacounty.gov" matches allowed_domains=["lacounty.gov"]).
    */
   export function isDomainAllowlisted(url: string, allowedDomains: string[] | null): boolean {
     if (!allowedDomains || allowedDomains.length === 0) return false;
     const host = hostFromUrl(url);
     if (!host) return false;
     return allowedDomains.some((d) => {
       const ad = d.toLowerCase().replace(/^www\./, '');
       return host === ad || host.endsWith('.' + ad);
     });
   }
   ```

**Add the orchestrator:**

```ts
/**
 * Run a complete discovery pass for one registered jurisdiction.
 *
 * Pipeline:
 *   1. Load the discovery_jurisdictions row. Throw if missing.
 *   2. Load essentials.races + race_candidates for this (jurisdiction_geoid, election_date)
 *      so we can (a) feed known races as agent context and (b) run the withdrawal diff.
 *   3. INSERT a discovery_runs row with status='running'. Keep its id.
 *   4. Try: call runDiscoveryAgent with the jurisdiction config + known races.
 *   5. For each discovered candidate: normalize, compute confidence, compute flagged state,
 *      find matched_candidate_id/race_id when possible, INSERT into candidate_staging.
 *   6. For each race the agent touched: diff existing candidates. Any existing candidate
 *      that doesn't fuzzy-match a discovered candidate >= 85% gets an action='withdrawal'
 *      staging row.
 *   7. UPDATE the discovery_runs row with status='completed', counts, raw_output, completed_at.
 *   8. On any thrown error after step 3: UPDATE run to status='failed', error_message = err.message.
 *      Re-throw.
 *
 * triggeredBy: optional label stored on the run ('on_demand' | 'cron'). Plan 05-04 passes 'on_demand'.
 */
export async function runDiscoveryForJurisdiction(
  discoveryJurisdictionId: string,
  opts: { triggeredBy?: string } = {}
): Promise<DiscoveryRunSummary> {
  // --- 1. Load config ---
  const cfgResult = await pool.query(
    `SELECT id, jurisdiction_geoid, jurisdiction_name, state, election_date,
            source_url, allowed_domains
       FROM essentials.discovery_jurisdictions
      WHERE id = $1`,
    [discoveryJurisdictionId]
  );
  if (cfgResult.rows.length === 0) {
    throw new Error(`[discoveryService] No discovery_jurisdictions row with id=${discoveryJurisdictionId}`);
  }
  const cfg = cfgResult.rows[0];

  // --- 2. Load known races + existing candidates for withdrawal diff ---
  // We key off (election_date). elections rows carry date + state; races link to elections.
  // This query is "best effort" — if schema links are weak we still proceed with empty knownRaces.
  const racesResult = await pool.query(
    `SELECT r.id AS race_id, r.position_name, r.primary_party
       FROM essentials.races r
       JOIN essentials.elections e ON e.id = r.election_id
      WHERE e.election_date = $1 AND e.state = $2`,
    [cfg.election_date, cfg.state]
  );
  const knownRaces = racesResult.rows.map((r: any) => ({
    race_id: r.race_id as string,
    position_name: r.position_name as string,
    primary_party: r.primary_party as string | null,
  }));

  const existingCandidatesResult = knownRaces.length
    ? await pool.query(
        `SELECT rc.id AS candidate_id, rc.race_id, rc.full_name
           FROM essentials.race_candidates rc
          WHERE rc.race_id = ANY($1::uuid[])`,
        [knownRaces.map((r) => r.race_id)]
      )
    : { rows: [] as any[] };
  const existingCandidates = existingCandidatesResult.rows.map((r: any) => ({
    id: r.candidate_id as string,
    raceId: r.race_id as string,
    fullName: r.full_name as string,
  }));

  // --- 3. Start run row ---
  const runInsert = await pool.query(
    `INSERT INTO essentials.discovery_runs
       (discovery_jurisdiction_id, jurisdiction_geoid, election_date, status, triggered_by)
     VALUES ($1, $2, $3, 'running', $4)
     RETURNING id`,
    [cfg.id, cfg.jurisdiction_geoid, cfg.election_date, opts.triggeredBy ?? 'on_demand']
  );
  const runId = runInsert.rows[0].id as string;

  try {
    // --- 4. Call agent ---
    const agentInput: DiscoveryAgentInput = {
      jurisdictionName: cfg.jurisdiction_name,
      state: cfg.state,
      electionDate: String(cfg.election_date).slice(0, 10),
      sourceUrl: cfg.source_url,
      allowedDomains: cfg.allowed_domains,
      knownRaces: knownRaces.map((r) => ({
        position_name: r.position_name,
        primary_party: r.primary_party,
      })),
    };
    const agentResult: DiscoveryAgentResult = await runDiscoveryAgent(agentInput);

    // --- 5. Stage each discovered candidate ---
    const allowedDomains = cfg.allowed_domains as string[] | null;
    let candidatesStaged = 0;
    const discoveredByRaceId = new Map<string, DiscoveredCandidate[]>();

    for (const cand of agentResult.candidates) {
      const domainOK = isDomainAllowlisted(cand.citation_url, allowedDomains);

      // Try to match the discovered name against existing candidates (for confidence + matched_candidate_id)
      let bestMatch: { candidateId: string; raceId: string; score: number } | null = null;
      for (const ex of existingCandidates) {
        const s = nameSimilarity(cand.full_name, ex.fullName);
        if (s >= NAME_MATCH_THRESHOLD && (!bestMatch || s > bestMatch.score)) {
          bestMatch = { candidateId: ex.id, raceId: ex.raceId, score: s };
        }
      }

      const confidence = scoreConfidence({
        domainOnAllowlist: domainOK,
        nameMatchesExistingCandidate: bestMatch !== null,
      });

      // race_id hint: if fuzzy match found, use its race. Otherwise try to infer from race_hint + position_name.
      let raceId: string | null = bestMatch?.raceId ?? null;
      if (!raceId) {
        const raceFromHint = knownRaces.find(
          (r) => nameSimilarity(r.position_name, cand.race_hint) >= NAME_MATCH_THRESHOLD
        );
        if (raceFromHint) raceId = raceFromHint.race_id;
      }

      // flagged? If we could not tie this to any race, flag it — that's the "missing race" radar.
      const flagged = raceId === null;
      const flagReason = flagged ? 'no matching race in DB' : null;

      await pool.query(
        `INSERT INTO essentials.candidate_staging
           (run_id, discovery_jurisdiction_id, full_name, normalized_name,
            citation_url, race_hint, race_id, matched_candidate_id,
            confidence, action, flagged, flag_reason, status)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9, 'new', $10, $11, 'pending')`,
        [
          runId,
          cfg.id,
          cand.full_name,
          normalizeName(cand.full_name),
          cand.citation_url,
          cand.race_hint,
          raceId,
          bestMatch?.candidateId ?? null,
          confidence,
          flagged,
          flagReason,
        ]
      );
      candidatesStaged++;

      if (raceId) {
        const arr = discoveredByRaceId.get(raceId) ?? [];
        arr.push(cand);
        discoveredByRaceId.set(raceId, arr);
      }
    }

    // --- 6. Withdrawal diff (only for races the agent touched) ---
    let withdrawalsStaged = 0;
    for (const [raceId, discovered] of discoveredByRaceId.entries()) {
      const existingInThisRace = existingCandidates.filter((ex) => ex.raceId === raceId);
      for (const ex of existingInThisRace) {
        const matched = discovered.some(
          (d) => nameSimilarity(d.full_name, ex.fullName) >= NAME_MATCH_THRESHOLD
        );
        if (!matched) {
          await pool.query(
            `INSERT INTO essentials.candidate_staging
               (run_id, discovery_jurisdiction_id, full_name, normalized_name,
                citation_url, race_hint, race_id, matched_candidate_id,
                confidence, action, flagged, flag_reason, status)
             VALUES ($1, $2, $3, $4, $5, $6, $7, $8, 'uncertain', 'withdrawal', true, $9, 'pending')`,
            [
              runId,
              cfg.id,
              ex.fullName,
              normalizeName(ex.fullName),
              // citation_url is NOT NULL in the schema; for a withdrawal we store the same
              // source_url the agent was pointed at (or a placeholder built from jurisdiction)
              // so the constraint holds. The review UI (Phase 6) will treat action='withdrawal'
              // differently from 'new' anyway.
              cfg.source_url ?? `about:withdrawal-diff/${cfg.jurisdiction_geoid}/${cfg.election_date}`,
              'withdrawal (diff)',
              raceId,
              ex.id,
              'no longer appears on official source',
            ]
          );
          withdrawalsStaged++;
        }
      }
    }

    // --- 7. Finalize run ---
    await pool.query(
      `UPDATE essentials.discovery_runs
          SET status = 'completed',
              completed_at = now(),
              candidates_found = $2,
              candidates_new = $3,
              candidates_withdrawn = $4,
              raw_output = $5::jsonb
        WHERE id = $1`,
      [
        runId,
        agentResult.candidates.length,
        candidatesStaged,
        withdrawalsStaged,
        JSON.stringify({
          model: agentResult.model,
          input_tokens: agentResult.inputTokens,
          output_tokens: agentResult.outputTokens,
          stop_reason: agentResult.stopReason,
          candidates: agentResult.candidates,
        }),
      ]
    );

    return {
      runId,
      jurisdictionId: cfg.id,
      candidatesFound: agentResult.candidates.length,
      candidatesStaged,
      withdrawalsStaged,
      status: 'completed',
      errorMessage: null,
    };
  } catch (err) {
    // --- 8. Mark run failed and re-throw ---
    const message = err instanceof Error ? err.message : String(err);
    await pool.query(
      `UPDATE essentials.discovery_runs
          SET status = 'failed', completed_at = now(), error_message = $2
        WHERE id = $1`,
      [runId, message]
    );
    throw err;
  }
}
```

**Avoid:**
- Do NOT wrap steps 4-6 in a Postgres transaction. If the agent succeeds but a single INSERT fails mid-loop, we want the partial data preserved + the run marked failed (for audit). Transactional rollback would hide what the agent found. The run row is the audit trail.
- Do NOT parallelize the staging INSERTs with Promise.all. Sequential keeps log output deterministic and avoids pool-exhaustion for jurisdictions with many candidates.
- Do NOT promote any staging row to race_candidates here. Approval is Plan 05-04; auto-upsert is deferred to Phase 7 (STAG-02).
- Do NOT diff races the agent did NOT find any candidate for. That was a 05-CONTEXT decision: absence of evidence is not evidence of withdrawal.
- Do NOT attempt withdrawal diff when the agent result has zero candidates total — the early return avoids the "agent ran but hit a server error page" case triggering mass false-positive withdrawals. If `agentResult.candidates.length === 0`, the `discoveredByRaceId` map is empty and the diff loop is a no-op (already true with the code above, no extra guard needed).
- Do NOT use the supabase client for ANY of these queries. The essentials schema is not in PostgREST.
- Do NOT call `normalizeName` on agent output more than once per candidate — once at INSERT time is enough; the `normalized_name` column is for future query use.
